275 int fatoracaoLU (MATRIZ *matriz, int *trocas) {
276 #ifdef DEBUG
277  fprintf(out, "[fatoracaoLU] Iniciando a fatoracao LU.\n");
278 #endif
279  double m;
280  for (int col = 0; col < matriz->tam-1; ++col) { // zerando colunas
281  if((trocas[col] = pivotamentoParcial(matriz, col)) == -1) { // se o pivotamento falha, a matriz é inválida
282  fprintf(stderr, "[fatoracaoLU] Matriz singular, impossivel fatorar.\n");
283  return -1;
284  }
285  trocas[matriz->tam-1] = matriz->tam-1;
286  for (int lin = col+1; lin < matriz->tam; ++lin) { // calcula o modificador m para cada linha
287  m = (matriz->dados[pos(lin, col, matriz->tam)]*1.0)/matriz->dados[pos(col, col, matriz->tam)];
288  // U
289  for (int j = col; j < matriz->tam; ++j) // percorrendo linha para aplicar m
290  matriz->dados[pos(lin, j, matriz->tam)] -= m*matriz->dados[pos(col, j, matriz->tam)];
291  // L
292  matriz->dados[pos(lin, col, matriz->tam)] = m; // armazena o modificador m onde seria 0
293  }
294  }
295 #ifdef DEBUG
296  fprintf(out, "[fatoracaoLU] Fatoracao LU completa.\n");
297 #endif

391 int calculaResiduo(MATRIZ A, MATRIZ inv_A, MATRIZ *R, int iter, double *tempoResiduo) {
392  unsigned int tam = A.tam;
393  double C, r = 0;
394  double tempo_r = timestamp();
395  for (int lin = 0; lin < tam; lin++) {
396  for (int col = 0; col < tam; col++) {
397  C = 0;
398  for (int k = 0; k < tam; k++)
399  C += A.dados[pos(lin,k,tam)]*inv_A.dados[pos(k,col,tam)];
400  C = (lin == col ? 1.0 - C : -C);
401  R->dados[pos(lin, col, tam)] = C; // R = I - A*inv_A
402  r += C*C; // ||r|| = sum(R[i,j]^2)
403  }
404  }
405  r = sqrt(r); // ||r|| = sqrt(sum(R[i,j]^2))
406  tempo_r = timestamp() - tempo_r;
407  *tempoResiduo += tempo_r;
408  if (r == N_INF || r == P_INF || r == NaN) {
409  #ifdef DEBUG
410  fprintf(out, "Residuo se tornou inf/NaN, iteracoes interrompidas.\n");
411  #endif
412  return 1;
413  }
414  fprintf(out, "# iter %d: %.17g\n", iter, r);
415  return 0;
