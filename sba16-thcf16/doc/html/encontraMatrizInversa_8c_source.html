227 #ifdef DEBUG
228  printf("[FATORACAOLU] Iniciando a fatoracao LU.\n");
229 #endif
230  double m;
231  for (int col = 0; col < matriz->tam-1; ++col) { // zerando colunas
232  if(pivotamentoParcial(matriz, col) == -1) { // se o pivotamento falha, a matriz é inválida
233  fprintf(stderr, "[fatoracaoLU] Matriz singular, impossivel fatorar.\n");
234  return -1;
235  }
236  for (int lin = col+1; lin < matriz->tam; ++lin) { // calcula o modificador m para cada linha
237  m = (matriz->dados[pos(lin, col, matriz->tam)]*1.0)/matriz->dados[pos(col, col, matriz->tam)];
238  // U
239  for (int j = col; j < matriz->tam; ++j) // percorrendo linha para aplicar m
240  matriz->dados[pos(lin, j, matriz->tam)] -= m*matriz->dados[pos(col, j, matriz->tam)];
241  // L
242  matriz->dados[pos(lin, col, matriz->tam)] = m; // armazena o modificador m onde seria 0
243  }
244  }
245 #ifdef DEBUG

247 #endif
