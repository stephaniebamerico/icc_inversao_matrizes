219 #ifdef DEBUG
220  printf("[FATORACAOLU] Iniciando a fatoracao LU.\n");
221 #endif
222  double m;
223  for (int col = 0; col < matriz->tam-1; ++col) { // zerando colunas
224  if(pivotamentoParcial(matriz, col) == -1) { // se o pivotamento falha, a matriz é inválida
225  fprintf(stderr, "[fatoracaoLU] Matriz singular, impossivel fatorar.\n");
226  return -1;
227  }
228  for (int lin = col+1; lin < matriz->tam; ++lin) { // calcula o modificador m para cada linha
229  m = (matriz->dados[pos(lin, col, matriz->tam)]*1.0)/matriz->dados[pos(col, col, matriz->tam)];
230  // U
231  for (int j = col; j < matriz->tam; ++j) // percorrendo linha para aplicar m
232  matriz->dados[pos(lin, j, matriz->tam)] -= m*matriz->dados[pos(col, j, matriz->tam)];
233  // L
234  matriz->dados[pos(lin, col, matriz->tam)] = m; // armazena o modificador m onde seria 0
235  }
236  }
237 #ifdef DEBUG

239 #endif
